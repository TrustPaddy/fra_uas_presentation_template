import numpy as np
from typing import Tuple

class RewardShaper:
    """Multi-objective reward function for trajectory tracking."""

    def __init__(self, w_track: float = 0.6, w_energy: float = 0.2,
                 w_safety: float = 0.2, collision_penalty: float = -50.0):
        self.w_track    = w_track        # Trajectory tracking weight
        self.w_energy   = w_energy       # Energy efficiency weight
        self.w_safety   = w_safety       # Safety constraint weight
        self.penalty    = collision_penalty

    def compute(self, state: np.ndarray, action: np.ndarray,
                next_state: np.ndarray) -> Tuple[float, dict]:
        # Tracking error (end-effector position)
        ee_error   = np.linalg.norm(state[:3] - self.target[:3])
        r_track    = -self.w_track * ee_error ** 2

        # Energy consumption (joint torque norm)
        r_energy   = -self.w_energy * np.linalg.norm(action) ** 2

        # Safety: collision & joint limit violations
        collision  = self._check_collision(next_state)
        r_safety   = self.penalty if collision else self.w_safety

        total      = r_track + r_energy + r_safety
        return total, {"track": r_track, "energy": r_energy, "safety": r_safety}

// PD-Regler mit Vorsteuerung
VectorXd Controller::compute(
    const VectorXd& q,
    const VectorXd& q_ref,
    const VectorXd& dq_ref) {

  VectorXd e    = q_ref - q;
  VectorXd de   = dq_ref - dq_;
  VectorXd tau  = M_ * (Kp_.cwiseProduct(e)
                       + Kd_.cwiseProduct(de))
                + C_ * dq_ref + g_;

  // Saturate torques
  return tau.cwiseMin(tau_max_)
            .cwiseMax(-tau_max_);
}
